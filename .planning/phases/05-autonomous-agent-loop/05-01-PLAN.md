# src/model/types.ts (pure types — no runtime imports)
export type ChatRole = 'system' | 'user' | 'assistant'
export type ChatContentPart =
  | { type: 'text'; text: string }
  | { type: 'image'; mediaType: string; dataBase64: string }   // base64 vision input
export interface ChatMessage { role: ChatRole; content: string | ChatContentPart[] }
export interface Provider { id: string; chat(messages: ChatMessage[]): Promise<string> }
export interface ModelSpec { provider: string; model: string }

# src/model/adapter.ts
export const DEFAULT_MODELS: { anthropic: string; scripted: string }   // as const string map
parseModelSpec(spec: string): ModelSpec
  // split on the FIRST ':'; 'anthropic:claude-haiku-4-5' → {provider,model}. A bare provider
  //   ('scripted') uses DEFAULT_MODELS[provider]. Unknown provider → throw a clear Error.
createProvider(spec: string, opts?: { apiKey?: string; baseUrl?: string; fetchImpl?: typeof fetch }): Provider
  // dispatch by provider. 'scripted' → createScriptedProvider(). 'anthropic' → requires opts.apiKey
  //   (throw a clear "set ANTHROPIC_API_KEY to use the anthropic provider" Error when missing).

# src/model/providers/anthropic.ts  (the ONLY file permitted to call fetch — GATE-03 v3)
export const ANTHROPIC_API_URL = 'https://api.anthropic.com/v1/messages'   // pinned endpoint constant
export const ANTHROPIC_VERSION = '2023-06-01'
buildAnthropicRequest(messages: ChatMessage[], model: string, baseUrl?: string):
  { url: string; headers: Record<string,string>; body: string }
  // PURE. Maps ChatMessage[] → Messages API JSON: 'system' messages → top-level `system` string;
  //   text parts → {type:'text',text}; image parts → {type:'image',source:{type:'base64',media_type,data}}.
  //   url = baseUrl ?? ANTHROPIC_API_URL. headers include 'anthropic-version' + 'x-api-key' PLACEHOLDER
  //   filled by the caller (the key is injected by createAnthropicProvider, never hard-coded here).
parseAnthropicResponse(json: unknown): string
  // PURE. Concatenate text blocks from content[]; throw a clear Error on an API error shape.
createAnthropicProvider(opts: { apiKey: string; model: string; baseUrl?: string; fetchImpl?: typeof fetch }): Provider
  // chat() = buildAnthropicRequest(...) → (opts.fetchImpl ?? fetch)(url,{method:'POST',headers,body})
  //   → parseAnthropicResponse(await res.json()). fetchImpl injection is what keeps the suite offline.

  //   The envelope schema is owned by the decision layer (05-02); this file only needs the two

createScriptedProvider(opts?: { seed?: number }): Provider
  // chat(messages) = read the last user message's embedded JSON envelope → decideScriptedAction →
  //   JSON.stringify(action). Deterministic. No fetch, no key, no imports outside src/model/ + node:.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| target page data → model provider | Screenshots + page summaries are sent to the user's chosen provider. This is the user's explicit BYO-key configuration (D5-01) and must be documented loudly — it is inherent to vision exploration. |
| provider endpoint → the internet | The single outbound surface must reach only the pinned, documented API host — no phone-home can hide behind it (endpoint pinning). |
| model layer ↔ capture/spec | `src/model/` must never read the capture store or the spec (import boundary) — the loop passes it observations; it must not be able to exfiltrate stored records. |
| injected API key ↔ source | The key is read at the CLI and injected; it is never hard-coded in a provider file. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | a provider quietly posting to a second, non-documented host (phone-home) | mitigate | GATE-03 v3 endpoint pin: the only https:// host literal permitted in a provider file is ANTHROPIC_API_URL; the only dynamic host is the user's explicit --model-base-url. Machine-checked in no-network.test.ts. |
| T-05-02 | Information Disclosure | the model layer reading the capture store / spec and shipping it out | mitigate | GATE-03 v3 import boundary: src/model/ imports nothing from src/capture/ or src/spec/; grep-enforced. The model only ever sees what the loop hands it. |
| T-05-03 | Spoofing / Elevation | outbound fetch leaking outside src/model/providers/ | mitigate | The bare-fetch + node:https exemption is scoped to the src/model/providers/ prefix only; every other src/ file keeps the original forbidden-token guard. |
| T-05-04 | Information Disclosure | the target's screenshots/summaries reaching a provider the user did not intend | accept-with-disclosure | This is the user's explicit configuration (no key → no autonomous mode). Documented loudly in CONTEXT + the CLI + the plan; not a code defect. |
| T-05-05 | Information Disclosure | an API key hard-coded or logged from a provider file | mitigate | The key is injected via opts.apiKey (read from ANTHROPIC_API_KEY at the CLI); anthropic.ts sets a placeholder header and never contains a key literal; parser/builder are pure and log nothing. |

No package installs in this plan — zero new runtime deps, no supply-chain checkpoint. The only new
outbound surface is the pinned provider endpoint, enforced by the evolved no-network guard.
